Redis messaging bus client.

Params
opts Object- Options for message bus.
[opts.host=127.0.0.1] String- Host of the redis server.
[opts.port=6379] Integer- Port that redis is running under on host.
[opts.password=''] String- Auth password for redis instance, if included Redis connections will be authenticated. Defaults to no password and therefor no authentication.
opts.namespace String- Namespace for messages. For two-way communication both clients must have same namespace set.
opts.clientId String- Identifier of this client, should be unique when compared to all other clients.
[opts.messageTimeout=30000] Number- The timeout length a message will wait for a response.
[opts.supressRedisError=true] Boolean- Whether the client should automatically reconnect to the redis server and ignore redis errors. Defaults to true.

Remus.prototype.listenToRoom = function(room, callback) {
  var roomc = room.replace(':', '??SC??');
  var roomChannel = this._namespace + ":room:" + roomc + ":*";

  if(!this._rooms.hasOwnProperty(roomChannel)) {
    this.incomingClient.psubscribe(roomChannel);
  }

  this._rooms[roomChannel] = true;

  return this.on('room:' + room, callback);
};

Remus.prototype.ignoreRoom = function(room, callback) {
  var roomc = room.replace(':', '??SC??');
  var roomChannel = this._namespace + ":room:" + roomc + ":*";

  delete this._rooms[roomChannel];

  if(typeof callback === 'Function') {
    return this.removeListener('room:' + room, callback);
  }
  return this.removeAllListeners('room:' + room);
};
